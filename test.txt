
@pytest.mark.parametrize("validation_errors,expected_messages", [
    # Case: Missing mandatory field
    (
        [
            {"type": "value_error.missing", "ctx": {"msg": "field missing"}},
        ],
        ["Mandatory field is missing."]
    ),
    # Case: Regex pattern mismatch
    (
        [
            {"type": "value_error.regex", "ctx": {"pattern": "[a-zA-Z]+"}},
        ],
        ["Invalid pattern. Expected pattern: [a-zA-Z]+"]
    ),
    # Case: Minimum length not met
    (
        [
            {"type": "value_error.any_str.min_length", "ctx": {"limit_value": 3}},
        ],
        ["Should have a minimum length of 3."]
    ),
    # Case: Enum validation failed
    (
        [
            {"type": "value_error.enum", "ctx": {"enum_values": ["Option1", "Option2"]}},
        ],
        ["Invalid value. Expected values: ['Option1', 'Option2']."]
    ),
    # Case: Default message for unknown error type
    (
        [
            {"type": "unknown_error", "ctx": {"msg": "An unknown error occurred"}},
        ],
        ["Error: An unknown error occurred"]
    ),
])
